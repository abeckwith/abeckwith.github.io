//create canvas to draw on: var canvas = document.getElementById("myCanvas");
//context object to call drawing methods on:
var context = canvas.getContext("2d");
//arrays for red squares (x and y loc)
var xrect = [20];
var yrect = [20];

canvas.width = 600;
canvas.height = 600;
canvas.addEventListener('click', clickReporter, false);
alert("Click close, then click anywhere in the black region");

//draws circles and squares
function draw_stuff(circleX, circleY, radius, fill)
{
     //clear the canvas
     context.clearRect(0, 0, canvas.width, canvas.height);
     context.beginPath();
     //draw background black rectangle:
     context.fillStyle = "black";
     context.fillRect(0, 0, canvas.width, canvas.height);
    
     for (i = 0; i < xrect.length; i++)
     {
          drawBox(xrect[i], yrect[i])
     }

     //speed at which x and y and radius change over time:
     if(x < canvas.width && x >= 0) x += xspeed; //move in x dirction
     else if(x < 0) x = canvas.width - 2; //wrap if off left side
     else x = radius; //wrap if go off right side

     if(y < canvas.height && y >= 0) y += yspeed; //move in y direction
     else if(y < 0) y = canvas.height - 2; //wrap if off top
     else y = radius; //wrap if off bottom

     //back to small radius if too big radius:
     if(r < 80) r += rspeed;
     else r = 1;
     //randomly change direction
     if(Math.random() < .01) xspeed = - xspeed;
          //randomly change direction
     if(Math.random() < .01) yspeed = - yspeed;
     //draw circle, given fill color sent to function:
     context.fillStyle = fill;
     //draws an arc with angle of 2PI, so whole circle:
     context.arc(circleX, circleY, radius, 0, Math.PI * 2);
     context.fillStyle = "green";
     context.arc(canvas.width - circleX, circleY, radius, 0, Math.PI * 2);
     context.fill();

}
function clickReporter(e)
{
     drawBox(e.clientX, e.clientY)
     //add x and y to arrays, so they can be displayed in the draw_circle function
     xrect.push(e.clientX)
     yrect.push(e.clientY)
} //draw a rectangle:
function drawBox(x, y)
{
     context.fillStyle = "red";
     context.fillRect(x - 20, y - 50, 20, 20);
}

//set initial values:
var x = 100;
var y = 100;
var r = 5;

     var xspeed = 5;
     var yspeed = 1;
     var rspeed = 0.1;
//timer calls draw circle repeatedly
setInterval(
     function() //function to execute:
     {
          draw_stuff(x, y, r , "cyan")
     }
     //time interval (msec)
     , 33
     );
